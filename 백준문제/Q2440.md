import java.util.Scanner;  
  
public class Q2440 {  
	public static void main(String[] args) {  
		Scanner sc = new Scanner(System.in);  
		int starNum = sc.nextInt();  
		String star = "";  
		for (int i = 0; i < starNum; i++) {  
			for (int j = 0; j < starNum - i; j++) {  
				star += "*";  
			}  
			System.out.println(star);  
			star = "";  
		}  
	}  
}
