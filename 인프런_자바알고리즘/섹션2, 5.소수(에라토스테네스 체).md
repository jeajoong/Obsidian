
소수 구하는법

// 1부터 n까지 배열을 이용해서 소수구하기
// 배열을 n +1 만큼 잡아서 자연수 체크 목적으로 배열 사용
// (0은 소수, 1은 소수 아님)

int[] ch = new int[n+1];
for(int i =2; i <= n; i++) {
	if(ch[i] == 0) {
		for(int j = i; j <= n; j = j+i) {
			ch[j] = 1;
		}
	}
}


// 특정숫자가 소수인지 확인하는 메서드
// (for문을 효율적으로 줄이는 방법을 강구해야할듯)
public boolean isPrime(int num) {
	if(num == 1) return false;
	for(int i = 2; i < num; i++) { // 2부터의 숫자로
		if(num % i == 0) {
			return false;
		}
	}
	return true;
}
